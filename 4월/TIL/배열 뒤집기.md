import java.io.*;
import java.util.*;

class Solution {
    static int N;
    public int[][] solution(int[][] image) {
        N = image.length;

        int pixel = 2*N;
        int[][] answer = new int[pixel][pixel];        
        
        int k = 0;
        for(int i=0; i<N; i++) {
            for(int j=0; j<N; j++) {
                answer[i][j] = image[i][j];
            }
            k = N - 1;
            for(int j=N; j<pixel; j++) {
                answer[i][j] = image[i][k];
                k--;
            }
        }

        int m = N;
        for(int i=N; i<pixel; i++) {
            m--;
            for(int j=0; j<pixel; j++) {
                answer[i][j] = answer[m][j];
            }
        }
        return answer;
    }
}
